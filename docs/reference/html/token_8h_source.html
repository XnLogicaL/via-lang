115struct Token {
116 TokenType type = TokenType::UNKNOWN;
117 std::string lexeme;
118 size_t line = 0, offset = 0, position = 0;
119
120 explicit Token() = default;
121 explicit Token(TokenType type, std::string lexeme, size_t line, size_t offset, size_t position)
122 : type(type),
123 lexeme(lexeme),
124 line(line),
125 offset(offset),
126 position(position) {}
127
128 // Returns the token as a string.
129 std::string to_string() const;
130
131 // Returns whether if the tokens lexeme is a value literal.
132 bool is_literal() const;
133
134 // Returns whether if the tokens lexeme is an operator.
135 bool is_operator() const;
136
137 // Returns whether if the tokens lexeme is a modifier.
138 bool is_modifier() const;
139
140 // Returns the operator precedence of the tokens lexeme or -1 if impossible.
141 int bin_prec() const;
